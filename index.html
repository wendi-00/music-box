    <script>

        const SCENES = ['scene1.png', 'scene2.png', 'scene3.png', 'scene4.png'];

        const PANDA_FILE = 'panda_character.png';

        const MUSIC_FILE = 'Jade_Echoes.mp3';


        const scene = new THREE.Scene();

        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);

        const renderer = new THREE.WebGLRenderer({ antialias: true });

        renderer.setSize(window.innerWidth, window.innerHeight);

        document.body.appendChild(renderer.domElement);


        const listener = new THREE.AudioListener();

        camera.add(listener);

        const sound = new THREE.Audio(listener);

        new THREE.AudioLoader().load(MUSIC_FILE, (buffer) => {

            sound.setBuffer(buffer);

            sound.setLoop(true);

            sound.setVolume(0.5);

        });


        document.getElementById('box-img').addEventListener('click', function() {

            if (sound.context.state === 'suspended') { sound.context.resume(); }

            sound.play(); 

            document.getElementById('box-overlay').style.opacity = '0';

            document.getElementById('box-overlay').style.transform = 'scale(2)';

            document.body.classList.add('active'); 

            setTimeout(() => { document.getElementById('box-overlay').style.display = 'none'; }, 1200);

        });


        const loader = new THREE.TextureLoader();

        const worldGroup = new THREE.Group();

        scene.add(worldGroup);


        SCENES.forEach((file, i) => {

            loader.load(file, (texture) => {

                const geometry = new THREE.CylinderGeometry(25, 25, 50, 40, 1, true, (Math.PI / 2) * i, Math.PI / 2);

                geometry.scale(-1, 1, 1);

                const material = new THREE.MeshBasicMaterial({ map: texture, side: THREE.DoubleSide });

                const panel = new THREE.Mesh(geometry, material);

                worldGroup.add(panel);

            });

        });


        // The Spinner Pedestal

        const pedestalGeom = new THREE.CylinderGeometry(4, 4.5, 1, 32);

        const pedestalMat = new THREE.MeshBasicMaterial({ color: 0x2a1a3d });

        const pedestal = new THREE.Mesh(pedestalGeom, pedestalMat);

        pedestal.position.y = -6;

        scene.add(pedestal);


        // The Character Figure

        let panda;

        loader.load(PANDA_FILE, (pTexture) => {

            panda = new THREE.Sprite(new THREE.SpriteMaterial({ map: pTexture }));

            panda.scale.set(12, 12, 1);

            panda.position.set(0, -1, 0);

            scene.add(panda);

        });


        camera.position.z = 18;


        let isDragging = false, prevX = 0;

        document.addEventListener('mousedown', () => isDragging = true);

        document.addEventListener('mouseup', () => isDragging = false);

        document.addEventListener('mousemove', (e) => {

            if (isDragging) { worldGroup.rotation.y += (e.clientX - prevX) * 0.005; }

            prevX = e.clientX;

        });


        function animate() {

            requestAnimationFrame(animate);

            const time = Date.now() * 0.002; 

            

            if (panda && document.body.classList.contains('active')) {

                // Ballerina Spin: Rotate the character on its own axis

                panda.rotation.y += 0.02; 

                // Light breathing/pulse

                panda.scale.set(12 + Math.sin(time) * 0.2, 12 + Math.sin(time) * 0.2, 1);

            }


            if (!isDragging && document.body.classList.contains('active')) {

                worldGroup.rotation.y += 0.002; 

            }

            renderer.render(scene, camera);

        }

        animate();

    </script>
